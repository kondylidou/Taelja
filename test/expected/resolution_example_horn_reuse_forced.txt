Axiom 1: s(Z)
Axiom 2: s(X) => q(b,X)
Axiom 3: s(X) => p(X)
Axiom 4: q(X,Y) /\ p(Y) => r(X,Y)

Lemma 11: p(Z)
Proof:
  have s(Z)
    by axiom 1
  hence p(Z)
    by axiom 3

Goal 1: r(b,a)
Proof:
  have s(a)
    by axiom 1
  hence q(b,a)
    by axiom 2
   and p(a)
    by lemma 11
  hence r(b,a)
    by axiom 4
