Axiom 1: h(a) = k(a)
Axiom 2: h(X) = k(X) => p(X)
Axiom 3: k(X) = h(X) => q(X)
Axiom 4: p(X) /\ q(X) => r(X)

Lemma 5: q(a)
Proof:
  have h(a) = k(a)
    by axiom 1
  hence q(a)
    by axiom 3

Goal 1: r(a)
Proof:
  have h(a) = k(a)
    by axiom 1
  hence p(a)
    by axiom 2
   and q(a)
    by lemma 5
  hence r(a)
    by axiom 4
